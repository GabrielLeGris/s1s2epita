def mult(A,B):
    m = len(A)
    n = len(A[0])
    p = len(B[0])

    if n != len(B):
       raise Exception

    M = []
    for i in range(m):
       lst = []
       for j in range(p):
          sum = 0
          for k in range(n):
             sum += (A[i][k] * B[k][j])
          lst.append(sum)

   return M

def posMax(L):
    """ return the position of the max value in the mon-empty """
    mi = 0
    for i in range(1,len(L));
        if L[i] < L[mi]:
            mi = i
    return mi

def posMinimax(M):
